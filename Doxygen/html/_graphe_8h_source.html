146 void Graphe<Ar, So>::copie(const Graphe<Ar, So> & graphe) {
147  Iterateur<Sommet<So>> itSommets = listeSommets.getIterateur();
148  itSommets.debut();
149 
150  // -------------- d'abord on recopie les sommets --------------------
151  const Sommet<So> * sommet;
152  while (itSommets.aSuivant()) {
153  sommet = itSommets.suivant(); // sommet courant à recopier
154  this->creeSommet(sommet->valeur); // on crée la copie du sommet courant avec le même identifiant
155  }
156 
157  // -------------------- et maintenant on recopie les aretes --------------------
158 
159  // attention, la recopie des aretes est plus compliquee car il faut rebrancher les aretes sur les nouveaux sommets qui viennent d'etre crees.
160  // Pour retrouver les "bons sommets" on utilise les clefs qui ont ete conservees
161 
162  Iterateur<Arete<Ar, So>> itAretes = listeAretes.getIterateur();
163  itAretes.debut();
164 
165  while (itAretes.aSuivant()) {
166  const Arete<Ar, So> * a = itAretes.suivant(); // arête courante à recopier
167  Sommet<So> * debut, *fin; // le début et la fin de la nouvelle arête qui va être créée
168  Liste< Sommet<So> > p;
169 
170  // on recherche d dans la nouvelle liste de sommets grâce à son identifiant
171  SommetDejaPresentDansLaCopie<So> conditionDebut(a->debut);
172  Liste<Sommet<So>> temp;
173  temp.tete = PElement< Sommet<So> >::appartient(listeSommets.tete, conditionDebut);
174  Iterateur<Sommet<So>> itTemp = temp.getIterateur();
175  while (itTemp.aSuivant())
176  p.ajouterElem(*itTemp.suivant());
177  debut = p.tete->valeur;
178 
179  // on recherche f dans la nouvelle liste de sommets grace a sa identifiant
180  SommetDejaPresentDansLaCopie<So> conditionFin(a->fin);
181  temp.tete = PElement< Sommet<So> >::appartient(listeSommets.tete, conditionFin);
182  itTemp = temp.getIterateur();
183  while (itTemp.aSuivant())
184  p.ajouterElem(*itTemp.suivant());
185  fin = p.tete->valeur;
186 
187  this->creeArete(a->valeur, debut, fin);
188  }
